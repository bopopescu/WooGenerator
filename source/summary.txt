from collections import OrderedDict
#abstract
class ImportObject(OrderedDict):
    def __init__(self, data):
        super(ImportObject, self).__init(data)

#tree
class ImportTreeObject(ImportObject):
    def __init__(self, data, tree_data):
        self.tree_data = tree_data
        super(ImportTreeObject, self).__init__(data)
        self.processMeta()

class ImportTreeRoot(ImportTreeObject):
    def __init__(self):
        data = OrderedDict()
        super(ImportTreeRoot, self).__init__(data)

class ImportTreeItem(ImportTreeObject):
    pass

class ImportTreeTaxo(ImportTreeObject):
    pass

#gen
class ImportGenObject(ImportTreeObject):
    def __init__(self, data, gen_data):
        self.gen_data = gen_data
